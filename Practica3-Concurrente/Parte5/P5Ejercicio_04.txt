Tres robots floreros tienen 8 intentos en total para juntar todas las flores 
dentro del cuadrante comprendido entre las esquinas (40,40) y (60,60). Para 
ello, en cada intento un robot fiscalizador indicar치 a un robot aleatorio la 
esquina a la que debe dirigirse. El fiscalizador calcular치 esta esquina de 
manera aleatoria. Al completarse los 8 intentos, los robots floreros deber치n 
depositar todas las flores juntadas en la esquina (10,10), y el robot 
fiscalizador deber치 informar la cantidad total de flores juntadas por los robots.
Los robots floreros inician en las esquinas (1,1), (2,1) y (3,1) respectivamente, 
y el fiscalizador en la (4,1).

programa P5Ej1;

procesos

	proceso enviarIds
	comenzar
		EnviarMensaje(1, robot1)
		EnviarMensaje(2, robot2)
		EnviarMensaje(3, robot3)
	fin 
	
	proceso elegirRobot(ES robot: numero)	
	comenzar
		Random(robot, 1, 3)
	fin
	
	proceso juntarFlores(ES flores: numero)
	comenzar
		mientras HayFlorEnLaEsquina
			tomarFlor
			flores:= flores + 1
	fin
	
	proceso dejarFlores(ES flores: numero)
	comenzar
		repetir flores
			depositarFlor
			flores:= flores - 1
	fin
	
	proceso elegirEsquina(ES av: numero; ES ca: numero)
	comenzar
		Random(av, 40, 60)
		Random(ca, 40, 60)
	fin
	
	proceso finalizarRobots
	comenzar
		EnviarMensaje(V, robot1)
		EnviarMensaje(V, robot1)
		EnviarMensaje(V, robot1)
	fin
	
areas
	areaFloresPC = AreaPC(40, 40, 60, 60)
	areaRobot1 = AreaP(1, 1, 1, 1)
	areaRobot2 = AreaP(2, 1, 2, 1)		
	areaRobot3 = AreaP(3, 1, 3, 1)		
	areaRobotFisco = AreaP(4, 1, 4, 1)
	
robots
	robot florero
	varibales
		id: numero
		av, ca: numero
		flores: numero
		finalizo: boolean
		
	comenzar
		avIni: PosAv
		caIni: PosCa
		
		RecibirMensaje(id, robotFisco)
		
		RecibirMensaje(finalizo)
		mientras ~finalizo
			RecibirMensaje(av, robotFisco)
			RecibirMensaje(ca, robotFisco)
			BloquearEsquina(av, ca)
			Pos(av, ca)
			juntarFlores(flores)
			Pos(avIni, caIni)
			LiberarEsquina(av, ca)
			RecibirMensaje(finalizo)
		
		BloquearEsquina(av, ca)
		Pos(av, ca)
		dejarFlores(flores)
		Pos(avIni, caIni)
		LiberarEsquina(av, ca)
		
	fin
	
	
	robot fiscalizador
	varibales
		nroRobot: numero
		av, ca: numero
		
	comenzar
		enviarIds
	
		repetir 8
			
			elegirEsquina(av, ca)			
			elegirRobot(nroRobot)
			
			si (nroRobot = 1)
				EnviarMensaje(V, robot1)
				EnviarMensaje(av, robot1)
				EnviarMensaje(ca, robot1)

					EnviarMensaje(ca, robot2)
					
				sino
					EnviarMensaje(V, robot3)
					EnviarMensaje(av, robot3)
					EnviarMensaje(ca, robot3)
		
		finalizarRobots
	fin
