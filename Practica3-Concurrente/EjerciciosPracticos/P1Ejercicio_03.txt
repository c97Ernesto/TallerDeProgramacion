
programa P1Ej3

procesos

	proceso apuntarArriba
	comenzar
		repetir 3
			derecha
	fin
	
	proceso juntarFlores(ES f: numero)
	comenzar
		mientras HayFlorEnLaEsquina
			tomarFlor
			f:= f + 1
	fin
	
	proceso dejarFlores(E f: numero)
	comenzar
		mientras HayFlorEnLaBolsa
			depositarFlor
	fin
	
	proceso juntarPapeles(ES p: numero)
	comenzar
		mientras HayPapelEnLaEsquina
			tomarPapel
			p:= p + 1
	fin
	
	proceso dejarPapeles(E p: numero)
	comenzar
		mientras HayPapelEnLaBolsa
			depositarPapel
	fin
	
	proceso Escalon(ES pasos: numero; ES mayorEnUno: numero)
	variables
		flores, papeles: numero
	comenzar
		papeles:= 0
		flores:= 0
		pasos:= pasos + 1
		
		repetir pasos
			mover
			juntarFlores(flores)
			dejarFlores(flores)
			juntarPapeles(papeles)
			dejarPapeles(papeles)
			
		derecha
		repetir pasos
			mover
			juntarFlores(flores)
			dejarFlores(flores)
			juntarPapeles(papeles)
			dejarPapeles(papeles)
			
		si (papeles = (flores + 1))
			mayorEnUno:= mayorEnUno + 1
	fin
	
	

areas
	areaC: AreaC(12, 6, 32, 24)

robots
	robot robocop
	variables
		pasos, mayorEnUno: numero
		
	comenzar
		mayorEnUno:= 0
		pasos:= 0
		
		repetir 4
			Escalon(pasos, mayorEnUno)
			apuntarArriba
			
		Informar('MayorEnUnoEnPapeles', mayorEnUno)
	fin
	

variables
	robot1: robocop
	robot2: robocop
	robot3: robocop

comenzar
	AsignarArea(robot1, areaC)
	AsignarArea(robot2, areaC)
	AsignarArea(robot3, areaC)
	Iniciar(robot1, 12, 14)
	Iniciar(robot2, 17, 10)
	Iniciar(robot3, 22, 6)
fin
